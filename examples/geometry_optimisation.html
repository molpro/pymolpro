assert p.status == 'completed'
method=p.xpath('//summary/@overall_method')[-1]

[14]:

def closest_interatomic(atoms):
    distances = atoms.get_all_distances()
    n = len(distances) // 2
    result = 1000.0
    for i in range(n):
        for j in range(n):
            result = min(distances[i][j + n], result)
    return result
