p=Project("molpro_variables")
p.write_input("""
do i=1,73
x(i)=i*10, cosine(i)=cos(x(i)), sine(i)=sin(x(i))
enddo
put,xml
""")
p.run(wait=True)
assert p.status=='completed' and not p.errors()
